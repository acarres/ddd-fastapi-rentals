    date_range = DateRange(
        start_date=date(2025, 1, 1),
        end_date=date(2025, 1, 31),

        start_date=date(2025, 1, 1),
        end_date=date(2025, 1, 31),

    date_range = DateRange(
        start_date=date(2025, 1, 1),

    )

    booking = Booking.create(date_range)
